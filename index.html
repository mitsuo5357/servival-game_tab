const touchPad = document.getElementById('touch-pad'); let camTouchId = null; let lastCamX = 0, lastCamY = 0; // ★ タップ判定用の変数を追加 let touchStartX = 0, touchStartY = 0, touchStartTime = 0; let isDragging = false; touchPad.addEventListener('pointerdown', (e) => { e.preventDefault(); if(camTouchId === null) { camTouchId = e.pointerId; lastCamX = e.clientX; lastCamY = e.clientY; // ★ タッチした場所と時間を記録する touchStartX = e.clientX; touchStartY = e.clientY; touchStartTime = Date.now(); isDragging = false; } }); touchPad.addEventListener('pointermove', (e) => { e.preventDefault(); if(e.pointerId === camTouchId) { const dx = e.clientX - lastCamX; const dy = e.clientY - lastCamY; // ★ 指が少しでも動いたら「スワイプ（視点移動）」だと判定する if (Math.abs(e.clientX - touchStartX) > 5 || Math.abs(e.clientY - touchStartY) > 5) { isDragging = true; } yaw += dx * 0.005; pitch += dy * 0.005; pitch = Math.max(-Math.PI/2 + 0.1, Math.min(Math.PI/2 - 0.1, pitch)); lastCamX = e.clientX; lastCamY = e.clientY; } }); function handleCamEnd(e) { if(e.pointerId === camTouchId) { camTouchId = null; // ★ 指を離した時、スワイプしてなくて短い時間（300ミリ秒以内）なら「壊す（タップ）」と判定！ if (!isDragging && (Date.now() - touchStartTime) < 300) { doAction(false); // false = 左クリック（壊す・叩く）と同じ動きを実行！ } } } touchPad.addEventListener('pointerup', handleCamEnd); touchPad.addEventListener('pointercancel', handleCamEnd);
